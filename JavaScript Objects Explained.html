JavaScript Objects Explained

Creating and working with objects in JavaScript

Creating Objects

Objects in JavaScript are collections of key-value pairs. Here are different ways to create objects:

Object Literal

let person = {
  firstName: "John",
  lastName: "Doe",
  age: 46,
  fullName: function() {
    return this.firstName + " " + this.lastName;
  }
};

new Object()

let person = new Object();
person.firstName = "Jane";
person.lastName = "Smith";
person.age = 32;
person.fullName = function() {
  return this.firstName + " " + this.lastName;
};

Constructor Function

function Person(first, last, age) {
  this.firstName = first;
  this.lastName = last;
  this.age = age;
  this.fullName = function() {
    return this.firstName + " " + this.lastName;
  }
}

let person = new Person("Bob", "Johnson", 28);

Accessing Object Properties

Dot Notation

let person = {
  firstName: "Alice",
  age: 25
};

let name = person.firstName;
let years = person.age;

Bracket Notation

let person = {
  firstName: "Alice",
  age: 25
};

let name = person["firstName"];
let years = person["age"];

Objects in Arrays

let students = [
  {name: "John", grade: 90},
  {name: "Sarah", grade: 95},
  {name: "Mike", grade: 85}
];

let firstStudent = students[0].name;
let secondGrade = students[1].grade;
