Assignment Operator

let x = 10; // Assigns value 10 to variable x
x += 5; // Equivalent to x = x + 5 → 15
x *= 2; // Equivalent to x = x * 2 → 30

Comparison Operators

Operator Name Example Result
== Equal to the value ? 5 == '5' true
=== Strict equal to the value and type ? 5 === '5' false
!= Not equal 5 != 3 true
!== Strict not equal 5 !== '5' true
> Greater than 10 > 5 true
< Less than 10 < 5 false
>= Greater than or equal 10 >= 10 true
<= Less than or equal 10 <= 5 false

Logical Operators

&&

Logical AND

(true && false) → false

||

Logical OR

(true || false) → true

!

Logical NOT

!true → false

Conditional Statements

if Statement

let age = 18;
if (age >= 18) {
  console.log("You're an adult");
}

else if Statement

let score = 85;
if (score >= 90) {
  grade = "A";
} else if (score >= 80) {
  grade = "B";
} else {
  grade = "C";
}

switch Statement

let day = 3;
switch(day) {
  case 1: dayName = "Monday"; break;
  case 2: dayName = "Tuesday"; break;
  case 3: dayName = "Wednesday"; break;
  default: dayName = "Invalid";
}

Break and Continue

break Statement

for (let i = 0; i < 10; i++) {
  if (i === 5) break;
  console.log(i);
}
// Output: 0,1,2,3,4

continue Statement

for (let i = 0; i < 5; i++) {
  if (i === 2) continue;
  console.log(i);
}
// Output: 0,1,3,4
